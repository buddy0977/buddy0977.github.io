EXP 1Ultrasonic & Buzzer
// Ultrasonic Sensor Pins
const int trigPin = 9;
const int echoPin = 10;
// Buzzer Pin
const int buzzer = 8;
long duration;
int distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  // Clear trigger pin
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  // Send ultrasonic pulse
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  // Read echo pulse
  duration = pulseIn(echoPin, HIGH);
  // Calculate distance
  distance = duration * 0.034 / 2;
  // Display distance
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");
  // Buzzer condition
  if(distance < 25) {
    digitalWrite(buzzer, HIGH);
  }
  else {
    digitalWrite(buzzer, LOW);
  }
  delay(200);
}

EXP 3DC Motor
// L298N Motor Driver Pins
const int ENA = 9;
const int IN1 = 8;
const int IN2 = 7;

void setup() {
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
}

void loop() {
  // Motor Forward Direction
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  // Speed Control using PWM
  analogWrite(ENA, 200);
  delay(3000);
  // Stop Motor
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  delay(1000);
  // Motor Reverse Direction
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  // Speed Control using PWM
  analogWrite(ENA, 150);
  delay(3000);
  // Stop Motor
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  delay(1000);
}

EXP 5DHT11 & Buzzer
#include <DHT.h>

// DHT11 setup
#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);
// Buzzer pin
const int buzzer = 8;
// Temperature threshold
float threshold = 30.0;

void setup() {
  Serial.begin(9600);
  dht.begin();
  pinMode(buzzer, OUTPUT);
}

void loop() {
  // Read temperature
  float temperature = dht.readTemperature();
  // Display temperature
  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.println(" °C");
  // Check threshold
  if (temperature > threshold) {
    digitalWrite(buzzer, HIGH);
  }
  else {
    digitalWrite(buzzer, LOW);
  }
  delay(2000);
}
EXP 7RTOS
#include <Arduino_FreeRTOS.h>

// LED Pins
const int redLED = 8;
const int greenLED = 9;
// Task Functions
void RedLEDTask(void *pvParameters);
void GreenLEDTask(void *pvParameters);

void setup() {
  pinMode(redLED, OUTPUT);
  pinMode(greenLED, OUTPUT);
  // Create Tasks
  xTaskCreate(
    RedLEDTask,
    "Red LED Task",
    128,
    NULL,
    1,
    NULL
  );
  xTaskCreate(
    GreenLEDTask,
    "Green LED Task",
    128,
    NULL,
    1,
    NULL
  );
}

void loop() {
  // Empty because FreeRTOS scheduler handles tasks
}

// Task for Red LED
void RedLEDTask(void *pvParameters) {
  while (1) {
    digitalWrite(redLED, HIGH);
    vTaskDelay(500 / portTICK_PERIOD_MS);
    digitalWrite(redLED, LOW);
    vTaskDelay(500 / portTICK_PERIOD_MS);
  }
}

// Task for Green LED
void GreenLEDTask(void *pvParameters) {
  while (1) {
    digitalWrite(greenLED, HIGH);
    vTaskDelay(1000 / portTICK_PERIOD_MS);
    digitalWrite(greenLED, LOW);
    vTaskDelay(1000 / portTICK_PERIOD_MS);
  }
}
